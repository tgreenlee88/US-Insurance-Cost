# comparing age vs cost 
# data is sorted by age youngest to oldest
def age_vs_charge():
    # graph of cost vs age
    sns.lineplot(x='age',y='charges',data=ip)
    
    sorted_ip = ip.sort_values(by='age')
    new_list = sorted_ip[["age","charges"]]
    print(new_list)
